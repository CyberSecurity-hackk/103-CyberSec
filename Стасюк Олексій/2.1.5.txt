#include <iostream>
using namespace std;
int main() {
	int a, b, c, d, e, year;
	cout << "Enter year`s number: ";
	cin >> year;
	a = year % 19;
	b = year % 4;
	c = year % 7;
	d = (a * 19 + 24) % 30;
	e = ((2 * b) + (4 * c) + (6 * d) + 5) % 7;
	cout << "Value = " << d + e << endl;
	if (e + d < 10)
		cout << "Easter falls on the March " << d + e + 22 << endl;
	else
		cout << "Easter falls on the April " << d + e - 9 << endl;
}
